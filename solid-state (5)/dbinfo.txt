CREATE TABLE ip_log (
	id INTEGER PRIMARY KEY AUTO_INCREMENT,
    visitor_ip TEXT NOT NULL,
		path TEXT NOT NULL,
    date TIMESTAMP
);

CREATE TABLE mail (
	id INTEGER PRIMARY KEY AUTO_INCREMENT,
		name TEXT NOT NULL,
    email TEXT NOT NULL, # No need for to - the emails are stored in the database
    message TEXT NOT NULL,
    ip TEXT NOT NULL,
		status TEXT NOT NULL,
    date TIMESTAMP
);
